#include<LiquidCrystal.h>
LiquidCrystal lcd(13,12,11,10,9,8);

#define Buzzer 4

void setup()
{
                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                
     Serial.begin(9600);
      lcd.begin(16,2);
  
      pinMode (Buzzer,OUTPUT);
      
      
      lcd.setCursor(0,0);
      lcd.print("    FISH   ");
      lcd.setCursor(0,1);
      lcd.print("MONITORING SYSTEM");
      delay(3000);
      lcd.clear();
}
void loop()
{
 
   if(Serial.available() )
         {
   char Y = Serial.read();
   Serial.print(Y);
    if (Y=='1')
         {
    digitalWrite (Buzzer,HIGH); 
   
    lcd.setCursor(0,0);
     lcd.print("Fish Identification              ");
    lcd.setCursor(0,1);
     lcd.print("Molly,Gold Panda               ");
  delay(1000);
     digitalWrite (Buzzer,LOW); 
Serial.print('A');
 delay(100);    
      }
 else if (Y=='2')
      {
         digitalWrite (Buzzer,LOW); 
   
    lcd.setCursor(0,0);
     lcd.print("NO Fish Identified               ");
         lcd.setCursor(0,1);
     lcd.print("                      ");
Serial.print('a');
 delay(100);
       }
            
 else
   {
          
digitalWrite (Buzzer,LOW); 
    }
      
  }
            
  }
